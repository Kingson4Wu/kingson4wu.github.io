+ 14个乒乓球分别贴上1-14数字，随机滚出4个，加起来是1001可能，其中11、12、13、14这个组合不算，剩下1000种可能。

func main() {

		nn = nn * i

		mm = mm * i

					//fmt.Printf("%d -> %d -> %d -> %d\n", i, j, k, l)

					count++

				index := hit

			}
			if round == 2 {

			}

			hit_map[result_rank] = 1
		}
